Button group

Groups a series of buttons together on a single line.

Bootstrap docs

Basic example

  • Result
  • HTML
  • Pug

<!-- Solid button group -->
<div class="btn-group" role="group" aria-label="Solid button group">
  <button type="button" class="btn btn-primary">Left</button>
  <button type="button" class="btn btn-primary">Middle</button>
  <button type="button" class="btn btn-primary">Right</button>
</div>

<!-- Outline button group -->
<div class="btn-group" role="group" aria-label="Outline button group">
  <button type="button" class="btn btn-outline-primary">Left</button>
  <button type="button" class="btn btn-outline-primary">Middle</button>
  <button type="button" class="btn btn-outline-primary">Right</button>
</div>
// Solid button group
.btn-group(role="group", aria-label="Solid button group")
  button(type="button").btn.btn-primary
    | Left
  button(type="button").btn.btn-primary
    | Middle
  button(type="button").btn.btn-primary
    | Right

// Outline button group
.btn-group(role="group", aria-label="Outline button group")
  button(type="button").btn.btn-outline-primary
    | Left
  button(type="button").btn.btn-outline-primary
    | Middle
  button(type="button").btn.btn-outline-primary
    | Right
